#include <string>
#include <vector>

using namespace std;

int solution(int n, int k) {
    int answer = 0;
    
    if(n>=10){ // 양꼬치가 10인분 이상인 경우
        answer = (12000 * n) + (2000 * (k-(n/10)));
    }
    else{ // 그 외인 경우
        answer = (12000 * n) + (2000 * k);
    }
    
    return answer;
}
