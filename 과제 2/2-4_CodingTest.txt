#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> answer;
    
    for(int i=0; i<flag.size(); i++){
        if(flag[i]){ // true : 끝에 arr[i]를 arr[i]*2개 만큼 넣는다.
            answer.insert(answer.end(), arr[i]*2, arr[i]); 
        } else{ // false : 시작위치(끝 - arr[i])부터 종료위치까지 제거한다.
            answer.erase(answer.end() - arr[i], answer.end());
        }
    }
    
    return answer;
}
