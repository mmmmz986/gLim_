#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    
    // 초기 좌표
    int minX = dots[0][0];
    int maxX = dots[0][0];
    int minY = dots[0][1];
    int maxY = dots[0][1];
    
    // 최대/최소 갱신
    for (int i=1; i<4; ++i){
        if(dots[i][0] < minX)
            minX = dots[i][0];
        if(dots[i][0] > maxX)
            maxX = dots[i][0];
        if(dots[i][1] < minY)
            minY = dots[i][1];
        if(dots[i][1] > maxY)
            maxY = dots[i][1];
    }
    int width = maxX - minX;
    int height = maxY - minY;
    answer = width * height;
    
    return answer;
}

int main(){
    vector<vector<int>> dots = {{1,1}, {2,1}, {2,2}, {1,2}};
    
    solution(dots);
    return 0;
}
